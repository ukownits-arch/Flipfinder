function addItem() { const name = document.getElementById("itemName").value; const price = document.getElementById("price").value; const file = document.getElementById("imageInput").files[0]; if (!name || !price || !file) { alert("Fill everything!"); return; } const reader = new FileReader(); reader.onload = function(e) { const image = e.target.result; const item = { name, price, image }; saveItem(item); displayItem(item); }; reader.readAsDataURL(file); } function displayItem(item) { const div = document.createElement("div"); div.className = "item"; div.innerHTML = `

${item.name}

£${item.price}

`; document.getElementById("listings").appendChild(div); } function saveItem(item) { const existing = JSON.parse(localStorage.getItem("listings")) || []; existing.push(item); localStorage.setItem("listings", JSON.stringify(existing)); } function loadItems() { const saved = JSON.parse(localStorage.getItem("listings")) || []; saved.forEach(item => displayItem(item)); } window.onload = loadItems;
